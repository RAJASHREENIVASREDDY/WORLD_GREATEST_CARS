01 — Form Validation

JavaScript validates user input before processing — name, email, and car selection all checked in real time.

Form Validation — JavaScript
function validateForm() {
  var name  = document.getElementById("name").value;
  var email = document.getElementById("email").value;
  var car   = document.getElementById("favCar").value;

  // Check if name is empty
  if (name === "") {
    alert("Please enter your name!");
    return false;
  }

  // Check email format using regex
  if (!email.includes("@")) {
    alert("Please enter a valid email!");
    return false;
  }

  alert("Welcome, " + name + "! You love " + car);
  return true;
}

03 — Favourites Counter (Variables + Events)

Click to add or remove from your favourites — JavaScript tracks count with a variable.

Counter using JS variables
var count = 0;  // Variable to track favourites

function addFavourite() {
  count++;  // Increment
  document.getElementById("count").textContent = count;
}

function removeFavourite() {
  if (count > 0) count--;  // Don't go below 0
  document.getElementById("count").textContent = count;
}
